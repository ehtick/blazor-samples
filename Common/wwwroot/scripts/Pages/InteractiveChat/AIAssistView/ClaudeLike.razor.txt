@code {
    private SfAIAssistView? sfAIAssistView;
    private IReadOnlyList<PromptResponseData> prompts = PromptResponseData.GetAllPromptResponseData();
    private bool isFirstPrompt = true;
    private string currentModel = "Opus 4.6";
    private bool? extendedThinkingEnabled = false;

    @*Hidden:Lines*@
    #pragma warning disable BL0005
    @*End:Hidden*@
    private AssistViewAttachmentSettings attachmentSettings = new AssistViewAttachmentSettings()
    {
        Enable = true,
        SaveUrl = "https://blazor.syncfusion.com/services/production/api/FileUploader/Save",
        RemoveUrl = "https://blazor.syncfusion.com/services/production/api/FileUploader/Remove"
    };
    @*Hidden:Lines*@
    #pragma warning restore BL0005
    @*End:Hidden*@

    private List<ModelData> modelItems = new()
    {
        new ModelData
        {
            Text = "Opus 4.6",
            Description = "Most capable for ambitious work"
        },

        new ModelData
        {
            Text = "Sonnet 4.6",
            Description = "Most efficient for everyday tasks"
        },

        new ModelData
        {
            Text = "Haiku 4.5",
            Description = "Fastest for quick answers"
        },

        new ModelData
        {
            Text = "Extended thinking",
            Description = "Think longer for complex tasks",
            Id = "extended-thinking"
        }
    };

    private async Task PromptRequest(AssistViewPromptRequestedEventArgs args)
    {
        if (isFirstPrompt)
        {
            isFirstPrompt = false;
            await InvokeAsync(StateHasChanged);
        }
        await Task.Delay(1200);

        var promptData = prompts.FirstOrDefault(p => p.Prompt == args.Prompt);
        var defaultResponse = "This is a placeholder Claude-style response. Connect to an LLM service for real output.";
        args.Response = promptData?.Response ?? defaultResponse;
    }

    private void OnModelSelected(MenuEventArgs args)
    {
        currentModel = args.Item.Text;
        StateHasChanged();
    }

    private void OnExtendedThinkingToggle(
        Syncfusion.Blazor.Buttons.ChangeEventArgs<bool?> args)
    {
        extendedThinkingEnabled = args.Checked;
    }

    private ModelData? GetModelData(string text)
    {
        return modelItems.FirstOrDefault(x => x.Text == text);
    }

    public class ModelData
    {
        public string Text { get; set; } = string.Empty;
        public string Description { get; set; } = string.Empty;
        public string Id { get; set; } = string.Empty;
    }
}
