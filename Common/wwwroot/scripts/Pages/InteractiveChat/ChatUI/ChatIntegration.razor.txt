<ActionDescription>
<p>This example demonstrates how to build a chat application using the Chat UI component. It illustrates customizing the properties and a header toolbar present inside the splitter layout that places the chat alongside a list view for straightforward navigation.</p>

<p>The interface supports multiple user conversations with dynamic switching between users; bot responses are triggered by user input and the chat updates to reflect the selected user's conversation. The layout is responsive, ensuring the experience adapts across screen sizes and devices.</p>
</ActionDescription>

@code {
    private string activeBreakpoint { get; set; } = "Large";
    private string HeaderText = "Albert";
    private string HeaderIconCss = "chat_user1_avatar";
    private string SaveURL = "https://blazor.syncfusion.com/services/production/api/FileUploader/Save";
    private string RemoveURL = "https://blazor.syncfusion.com/services/production/api/FileUploader/Remove";
    private UserModel User = new UserModel();
    private List<ChatMessage> Messages = new List<ChatMessage>();
    private List<string> ChatSuggestions = new List<string>();
    private SfChatUI integrationSample = new SfChatUI();
    private SfListView<ListViewData> ListViewObj = new SfListView<ListViewData>();
    private List<ListViewData> ListItems = ChatMessagesData.GetListItems().ToList();
    private List<ChatMessage> chatMessage1 = ChatMessagesData.GetChatMessages()[0].ToList();
    private List<ChatMessage> chatMessage2 = ChatMessagesData.GetChatMessages()[1].ToList();
    private List<ChatMessage> chatMessage3 = ChatMessagesData.GetChatMessages()[2].ToList();
    private List<ChatMessage> chatMessage4 = ChatMessagesData.GetChatMessages()[3].ToList();
    private List<ChatMessage> chatMessage5 = ChatMessagesData.GetChatMessages()[4].ToList();
    private List<ChatMessage> chatMessage6 = ChatMessagesData.GetChatMessages()[5].ToList();
    private List<TemplateMessageData> botData = ChatMessagesData.GetBotData().ToList();
    private Dictionary<string, object> ListOfChatMessages = new Dictionary<string, object>();
    private bool isToggleVisible { get; set; } = true;
    private List<SfChatUI> ListChats = new List<SfChatUI>();

    protected override void OnInitialized()
    {
        base.OnInitialized();
        User = ChatMessagesData.GetSpecificUser("user1", "Albert", "andrew", string.Empty);
        Messages = chatMessage1;
        ListChats = new List<SfChatUI>()
        {
            @*Hidden:Lines*@
            #pragma warning disable BL0005
            @*End:Hidden*@
            new SfChatUI()
            {
                HeaderIconCss = "chat_user1_avatar",
                HeaderText = "Albert",
                User = ChatMessagesData.GetSpecificUser("user1", "Albert", "andrew", string.Empty),
                Messages = chatMessage1
            },
            new SfChatUI()
            {
                HeaderIconCss = "chat_bot_avatar",
                HeaderText = "Decor bot",
                User =  ChatMessagesData.GetSpecificUser("admin", "Admin", "bot", string.Empty),
                Messages = chatMessage2,
                Suggestions = ChatMessagesData.GetBotSuggestions().ToList()
            },
            new SfChatUI()
            {
                HeaderIconCss = "chat_user2_avatar",
                HeaderText = "Charlie",
                User =  ChatMessagesData.GetSpecificUser("user2", "Charlie", "charlie", string.Empty),
                Messages = chatMessage3
            },
            new SfChatUI()
            {
                HeaderIconCss = "chat_user3_avatar",
                HeaderText = "Laura",
                User =  ChatMessagesData.GetSpecificUser("user3", "Laura", "laura", string.Empty),
                Messages = chatMessage4
            },
            new SfChatUI()
            {
                HeaderIconCss = "chat_team_avatar",
                HeaderText = "New Dev Team",
                User =  ChatMessagesData.GetSpecificUser("team", "Admin", "calendar", string.Empty),
                Messages = chatMessage5
            },
            new SfChatUI()
            {
                HeaderIconCss = "chat_user4_avatar",
                HeaderText = "Reena",
                User =  ChatMessagesData.GetSpecificUser("user4", "Albert", string.Empty, string.Empty),
                Messages = chatMessage6
            }
            @*Hidden:Lines*@
            #pragma warning restore BL0005
            @*End:Hidden*@
        };
        ListOfChatMessages = new Dictionary<string, object>()
        {
            { "user1", chatMessage1 },
            { "admin", chatMessage2 },
            { "user2", chatMessage3 },
            { "user3", chatMessage4 },
            { "team", chatMessage5 },
            { "user4", chatMessage6 }
        };
    }
    private void UpdateNextChat(ClickEventArgs<ListViewData> args)
    {
        ListOfChatMessages[integrationSample.User.ID] = integrationSample.Messages;
        integrationSample.Suggestions?.Clear();
        var newChat = ListChats[args.Index];
        User = newChat.User;
        HeaderIconCss = newChat.HeaderIconCss;
        HeaderText = newChat.HeaderText;
        Messages = newChat.Messages;
        if (newChat.User.ID == "admin") { ChatSuggestions = ChatMessagesData.GetBotSuggestions().ToList(); }
        UpdateToggle();
    }
    private void MessageSend(ChatMessageSendEventArgs args)
    {
        if (args.Message.Author.ID == "admin")
        {
            var foundMessage = botData.FirstOrDefault((message) => message.Text == args.Message.Text);
            var defaultResponse = "Use any real-time data streaming service to provide chat updates.";
            var message = new ChatMessage()
            {
                Author = ChatMessagesData.GetSpecificUser("bot", "Bot", "bot", String.Empty),
                Text = foundMessage != null ? foundMessage.Reply : defaultResponse
            };
            integrationSample.Messages.Add(message);
            ChatSuggestions = foundMessage != null ? foundMessage.Suggestions.ToList() : new List<string>();
        }
    }

    private void UpdateToggle()
    {
        if (activeBreakpoint == "Small")
        isToggleVisible = !isToggleVisible;
    }
}
