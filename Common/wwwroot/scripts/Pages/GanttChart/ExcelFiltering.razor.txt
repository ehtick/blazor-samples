    <p>
        When the <a class="e-gantt-excelfiltering-reference-link" aria-label="Navigate to the AllowFiltering property reference for gantt chart component"
                    target='_blank' href="https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.Gantt.SfGantt-1.html#Syncfusion_Blazor_Gantt_SfGantt_1_AllowFiltering" rel="noopener">AllowFiltering</a>

                                   target='_blank' href="https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.Gantt.GanttFilterSettings.html#Syncfusion_Blazor_Gantt_GanttFilterSettings_FilterType" rel="noopener">FilterType</a> 
        is set to <a class="e-gantt-excelfiltering-reference-link" aria-label="Navigate to the Excel filtertype reference for gantt chart component" target='_blank' href="https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.Gantt.FilterType.html#Syncfusion_Blazor_Gantt_FilterType_Excel" rel="noopener">Excel</a>,
        column headers display filtering menus that support condition based filtering similar to Excel. This allows precise control over which records are displayed.
    </p>
    <p>
        The <a class="e-gantt-excelfiltering-reference-link" aria-label="Navigate to the HierarchyMode property reference for gantt chart filter settings" target='_blank' href="https://help.syncfusion.com/cr/blazor/Syncfusion.Blazor.Gantt.GanttFilterSettings.html#Syncfusion_Blazor_Gantt_GanttFilterSettings_HierarchyMode" rel="noopener">HierarchyMode</a> property defines how related tasks are shown during filtering:
    </p>
    <ul>
        <li><code>Parent</code> - Displays filtered records along with their parent tasks. If no parent exists, only the filtered records are shown.</li>
        <li><code>Child</code> - Displays filtered records along with their child tasks. If no child exists, only the filtered records are shown.</li>
        <li><code>Both</code> - Displays filtered records along with both parent and child tasks. If no related tasks exist, only the filtered records are shown.</li>
        <li><code>None</code> - Displays only the filtered records without any hierarchical context.</li>
    </ul>
    <p>
        These options provide flexibility in how task relationships are visualized during filtering.
    </p>
    <p>
        For further details, refer to the <a class="e-gantt-excelfiltering-reference-link" aria-label="Navigate to the documentation for filtering in gantt chart component" target="_blank" rel="noopener" href="https://blazor.syncfusion.com/documentation/gantt-chart/excel-like-filter">documentation</a>.
    </p>

@code{
    protected override void OnInitialized()
	{
        if (SampleService != null) 
        { 
		    SampleService.TouchMousePreferenceChanged += OnTouchMousePreferenceChanged;
        }
	}

    private async void OnTouchMousePreferenceChanged()
    {
        if (GanttInstance == null)
        {
            return;
        }
        await GanttInstance.RefreshAsync();
        await GanttInstance.CallStateHasChangedAsync();
    }
#if WASM
	private bool _disposed;
	protected override async Task OnAfterRenderAsync(bool firstRender)
	{
		if (!firstRender || _disposed || GanttInstance == null)
		{
			return;
		}

		await Task.Delay(30);

		if (_disposed || GanttInstance == null)
		{
			return;
		}

		await GanttInstance.RefreshAsync().ConfigureAwait(true);
	}
#endif
    public void Dispose()
    {
#if WASM
		_disposed = true;
#endif
        SampleService.TouchMousePreferenceChanged -= OnTouchMousePreferenceChanged;
    }
}
