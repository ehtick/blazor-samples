@code {
	protected override void OnInitialized()
	{
		if (SampleService != null)
		{
			SampleService.TouchMousePreferenceChanged += OnTouchMousePreferenceChanged;
		}
	}
	
	private async void OnTouchMousePreferenceChanged()
	{
		if (GanttChartInstance == null)
		{
			return;
		}
		await GanttChartInstance.RefreshAsync();
		await GanttChartInstance.CallStateHasChangedAsync();
	}
#if WASM
	private bool _disposed;
	protected override async Task OnAfterRenderAsync(bool firstRender)
	{
		if (!firstRender || _disposed || GanttChartInstance == null)
		{
			return;
		}

		await Task.Delay(30);

		if (_disposed || GanttChartInstance == null)
		{
			return;
		}

		await GanttChartInstance.RefreshAsync().ConfigureAwait(true);
	}
#endif
	public void Dispose()
	{
#if WASM
		_disposed = true;
#endif
		SampleService.TouchMousePreferenceChanged -= OnTouchMousePreferenceChanged;
	}
	
}
