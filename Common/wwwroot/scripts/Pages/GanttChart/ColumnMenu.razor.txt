@code {
	protected override void OnInitialized()
	{
		if (SampleService != null) 
		{ 
			SampleService.TouchMousePreferenceChanged += OnTouchMousePreferenceChanged;
		}
	}
	
	private async void OnTouchMousePreferenceChanged()
	{
		if (GanttInstance == null)
		{
			return;
		}
		await GanttInstance.RefreshAsync();
		await GanttInstance.CallStateHasChangedAsync();
	}
#if WASM
	private bool _disposed;
	protected override async Task OnAfterRenderAsync(bool firstRender)
	{
		if (!firstRender || _disposed || GanttInstance == null)
		{
			return;
		}

		await Task.Delay(30);

		if (_disposed || GanttInstance == null)
		{
			return;
		}

		await GanttInstance.RefreshAsync().ConfigureAwait(true);
	}
#endif
	public void Dispose()
	{
#if WASM
		_disposed = true;
#endif
		SampleService.TouchMousePreferenceChanged -= OnTouchMousePreferenceChanged;
	}

}
